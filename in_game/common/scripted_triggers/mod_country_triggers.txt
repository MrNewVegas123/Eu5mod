REPLACE:can_upgrade_country_rank_empire = {
	is_great_power = yes
	trigger_if = {
		limit = { 
			country_exists = c:PAP
			religion = religion:catholic 
		}
		c:PAP = {
			opinion = { target = root value > 100 }
		}
	}
	OR = {
		total_population > 10000
		total_true_faith_population > 7500
		total_accepted_culture_population > 5000
	}
	prestige > 30
	is_subject = no
}
